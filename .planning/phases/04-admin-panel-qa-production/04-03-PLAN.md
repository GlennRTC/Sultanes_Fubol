handleBlockToggle(userId: string, currentBlocked: boolean):
  setActionLoading(userId); setError(''); setSuccess('');
  const { error: rpcErr } = await supabase.rpc('admin_block_user', {
    p_target_user_id: userId,
    p_blocked: !currentBlocked,
  });
  if (rpcErr) {
    setError(rpcErr.message === 'not_admin'
      ? 'No tienes permisos de administrador.'
      : 'Error al actualizar usuario. Intenta de nuevo.');
  } else {
    setSuccess(!currentBlocked ? 'Usuario bloqueado.' : 'Usuario desbloqueado.');
    // Update local state to reflect new is_blocked value:
    setUsers(prev => prev.map(u => u.id === userId ? { ...u, is_blocked: !currentBlocked } : u));
  }
  setActionLoading(null);

handleGrantTokens(e: React.FormEvent):
  e.preventDefault();
  const amount = parseInt(grantAmount, 10);
  if (isNaN(amount) || amount === 0) { setError('Ingresa un monto válido (distinto de cero).'); return; }
  if (!grantUserId) { setError('Selecciona un usuario.'); return; }
  setActionLoading(grantUserId); setError(''); setSuccess('');
  const { error: rpcErr } = await supabase.rpc('admin_grant_tokens', {
    p_target_user_id: grantUserId,
    p_amount: amount,
    p_note: grantNote,
  });
  if (rpcErr) {
    const msg = rpcErr.message;
    setError(
      msg === 'not_admin' ? 'No tienes permisos de administrador.' :
      msg === 'user_not_found' ? 'Usuario no encontrado.' :
      msg === 'amount_cannot_be_zero' ? 'El monto no puede ser cero.' :
      msg === 'insufficient_tokens' ? 'El usuario no tiene suficientes fichas para remover esa cantidad.' :
      'Error al actualizar fichas. Intenta de nuevo.'
    );
  } else {
    setSuccess(`${amount > 0 ? '+' : ''}${amount} fichas aplicadas a ${grantUserId}.`);
    // Refresh users to reflect new token balance:
    const { data } = await supabase.from('profiles').select('*').order('username');
    if (data) setUsers(data);
    setGrantAmount(''); setGrantNote('');
  }
  setActionLoading(null);

handleResetPassword(userId: string, username: string):
  setActionLoading(userId); setError(''); setSuccess('');
  const { error: fnErr } = await supabase.functions.invoke('admin-reset-password', {
    body: { userId },
  });
  if (fnErr) {
    setError('No se pudo enviar el correo de restablecimiento. Intenta de nuevo.');
  } else {
    setSuccess(`Correo de restablecimiento enviado para ${username}.`);
  }
  setActionLoading(null);

handleSetScore(matchId: string):
  const s = scores[matchId];
  const home = parseInt(s?.home ?? '', 10);
  const away = parseInt(s?.away ?? '', 10);
  if (isNaN(home) || isNaN(away) || home < 0 || away < 0) {
    setError('Ingresa marcadores válidos (números enteros >= 0).');
    return;
  }
  setSubmitting(matchId); setError(''); setSuccess('');
  const { error: rpcErr } = await supabase.rpc('admin_set_match_result', {
    p_match_id: matchId,
    p_home_score: home,
    p_away_score: away,
  });
  if (rpcErr) {
    setError(
      rpcErr.message === 'not_admin' ? 'No tienes permisos de administrador.' :
      rpcErr.message === 'match_not_found' ? 'Partido no encontrado.' :
      rpcErr.message === 'invalid_score' ? 'Los marcadores deben ser >= 0.' :
      'Error al registrar resultado. Intenta de nuevo.'
    );
  } else {
    setSuccess('Resultado registrado. Los puntos de quiniela se actualizaron automáticamente.');
    // Remove the match from the list (it is now 'finished' after calculate_prediction_points runs):
    setMatches(prev => prev.filter(m => m.id !== matchId));
  }
  setSubmitting(null);

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser→AdminRoute | UI guard checks profile.is_admin; real security enforced by SQL functions |
| browser→admin RPC | All five admin functions re-validate is_admin server-side on every call |
| browser→Edge Function | Authorization header carries user session JWT; Edge Function validates and re-checks is_admin |
| Edge Function→service role client | SUPABASE_SERVICE_ROLE_KEY lives only in Deno.env; never returned to browser |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-12 | Elevation of Privilege | /admin/* accessed by non-admin authenticated user | mitigate | AdminRoute redirects to /calendario; SQL functions reject with 'not_admin' as defense-in-depth |
| T-04-13 | Information Disclosure | Service role key exposed via Edge Function response | mitigate | adminClient is constructed inside Edge Function using Deno.env; key is never included in response body or headers |
| T-04-14 | Spoofing | JWT forged to bypass admin check in Edge Function | mitigate | anonClient.auth.getUser() validates JWT signature against Supabase signing key; client-supplied JWTs cannot be forged |
| T-04-15 | Tampering | Token grant with no audit log | mitigate | admin_grant_tokens SECURITY DEFINER function inserts to admin_logs inside the same transaction; if log INSERT fails, token update rolls back |
| T-04-16 | Elevation of Privilege | AdminRoute guard bypassed via direct URL | mitigate | AdminRoute is inside the outer ProtectedRoute; unauthenticated → /login before AdminRoute is reached |
| T-04-17 | Repudiation | Password reset action not logged | mitigate | Edge Function inserts 'password_reset_sent' to admin_logs via service role client after generateLink succeeds |
| T-04-SC | Tampering | npm/pip/cargo installs | accept | No new npm packages; Edge Function uses Deno import from esm.sh (same CDN pattern as existing ping function) |
</threat_model>
